// array of pointer to structure 
#include <stdio.h>
struct book
{ int id;
    char title[20];
    char author[20];};
int main(){
struct book b1={101,"c-intro","ritche"};
struct book b2={102,"story","AAA"};
struct book b3={103,"fab","XXX"};
//array of pointer to structure
struct book*books[3]={&b1,&b2,&b3};
int searchid,i,found=0;
printf("enter book id to search");
scanf("%d",&searchid);
for(i=0;i<8;i++){
    if(books[i]->id==searchid){
        found=1;
        printf("book found\n");
        printf("id=%d\n",books[i]->id);
        printf("title=%s\n",books[i]->title);
        printf("author=%s\n",books[i]->author); break;  }}
   if(!found)
   printf("not found\n");
    return 0;}
