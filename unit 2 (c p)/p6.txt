// program for binary search
#include <stdio.h>

int main() {
    int key;
    int mid;
    int arr[5]={10,20,30,40,50};
    printf("enter an element to search");
    scanf("%d",&key);
    int low =0, high =5;
     while(low<=high)
  {mid = (low +high)/2;
  if(arr[mid]==key)
{ printf("appeared at index %d",mid);
return 0;
}
else if(arr[mid]<key)
low=mid+1;
else
high=mid-1;}
printf(" not appeared at index ");
    return 0;
}
